# "How to make a simple calculator in Python."

first = float(input("First Number: "))
second = float(input("Second Number: "))
print("Addition")
sum = first + second
print(sum)
print("Subtraction")
sum = first - second
print(sum)
print("Multiplication")
sum = first * second
print(sum)
print("Division")
sum = first / second
print(sum)
